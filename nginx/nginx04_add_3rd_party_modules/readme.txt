// 开始操作
// 下载luajit
[root@nginx7server download]# wget http://luajit.org/download/LuaJIT-2.0.5.tar.gz
[root@nginx7server download]# tar -xvf LuaJIT-2.0.5.tar.gz
[root@nginx7server download]# cd LuaJIT-2.0.5/
